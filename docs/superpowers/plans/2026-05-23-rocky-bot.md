**Goal:** Build `rocky-bot`, a Python daemon that pulls Binance spot prices and places small ($30–80 notional) HMAC-signed orders against `/fapi/v1` on two accounts (MM + Taker) to keep `demo.rocky.exchange/perp/{BTC-PERP,ETH-PERP}` showing continuous order-book depth + fills.

Deployed to EC2 as a systemd-user unit.

```python
from pydantic import Field, field_validator
from pydantic_settings import BaseSettings, SettingsConfigDict


class Settings(BaseSettings):
    rocky_fapi_url: str
    mm_api_key: str
    mm_api_secret: str
    taker_api_key: str
    taker_api_secret: str
    symbols: list[str] = Field(default=["BTC-PERP", "ETH-PERP"])
    log_verbose: bool = False

    model_config = SettingsConfigDict(
        env_file=".env",
        env_file_encoding="utf-8",
        case_sensitive=False,
        extra="ignore",
    )

    @field_validator("symbols", mode="before")
    @classmethod
    def _split_csv(cls, v):
        if isinstance(v, str):
            return [s.strip() for s in v.split(",") if s.strip()]
        return v
```

```python
"""HMAC-SHA256 signer for /fapi requests (Binance Futures convention).

Caller composes the param dict in the order it wants sent (insertion order
preserved on Python 3.7+). `build_query` urlencodes; `sign` HMACs the
resulting string.
"""

import hashlib
import hmac
from typing import Any
from urllib.parse import quote


def build_query(params: dict[str, Any]) -> str:
    """Encode a dict to a URL query string. Skips None values."""
    parts = []
    for k, v in params.items():
        if v is None:
            continue
        parts.append(f"{quote(str(k), safe='')}={quote(str(v), safe='')}")
    return "&".join(parts)


def sign(secret: str, payload: str) -> str:
    """Return lowercase hex HMAC-SHA256(secret, payload)."""
    return hmac.new(secret.encode("utf-8"), payload.encode("utf-8"), hashlib.sha256).hexdigest()
```

```python
    ("BTCUSDT", "BTC-PERP"),
    ("ETHUSDT", "ETH-PERP"),
]


def binance_to_rocky(binance: str) -> str | None:
    for b, r in _PAIRS:
        if b == binance:
            return r
    return None


def rocky_to_binance(rocky: str) -> str | None:
    for b, r in _PAIRS:
        if r == rocky:
            return b
    return None


def all_pairs() -> list[tuple[str, str]]:
    return list(_PAIRS)
```

- [ ] **Step 5.2: Implement rocky_client.py**

```python
"""Async /fapi client.

Public methods cover the 8 endpoints rocky-bot uses. Each signed call:
  1. Builds params dict including timestamp=now_ms.
  2. Encodes to query string.
  3. Computes HMAC-SHA256 over the query.
  4. Appends signature= to the query (NEVER to the body).
  5. Sends with X-MBX-APIKEY header.

Returns parsed JSON on 2xx; raises FapiError(code, msg) on non-2xx.
"""

import time
from typing import Any

import httpx

from .sign import build_query, sign


class FapiError(Exception):
    def __init__(self, code: int, msg: str):
        super().__init__(f"[{code}] {msg}")
        self.code = code
        self.msg = msg


def _now_ms() -> int:
    return int(time.time() * 1000)


class RockyClient:
    def __init__(
        self,
        httpc: httpx.AsyncClient | None = None,
        api_key: str = "",
        api_secret: str = "",
        base_url: str | None = None,
    ):
        if httpc is None:
            assert base_url, "base_url required if httpc not provided"
            httpc = httpx.AsyncClient(base_url=base_url, timeout=10.0)
        self._http = httpc
        self._key = api_key
        self._secret = api_secret

    async def aclose(self):
        await self._http.aclose()

    # ── helpers ─────────────────────────────────────────────────────────────

    def _signed_query(self, params: dict[str, Any]) -> str:
        merged = {**params, "timestamp": _now_ms()}
        q = build_query(merged)
        sig = sign(self._secret, q)
        return f"{q}&signature={sig}"

    async def _signed_request(
        self, method: str, path: str, params: dict[str, Any]
    ) -> Any:
        query = self._signed_query(params)
        url = f"{path}?{query}"
        resp = await self._http.request(
            method, url, headers={"X-MBX-APIKEY": self._key}
        )
        if resp.status_code >= 300:
            try:
                j = resp.json()
                raise FapiError(int(j.get("code", -1000)), str(j.get("msg", "unknown")))
            except (ValueError, KeyError):
                raise FapiError(-1000, f"http {resp.status_code}")
        return resp.json()

    # ── public endpoints (no auth) ──────────────────────────────────────────

    async def exchange_info(self) -> Any:
        resp = await self._http.get("/fapi/v1/exchangeInfo")
        resp.raise_for_status()
        return resp.json()

    async def ticker_price(self, symbol: str | None = None) -> Any:
        params = {"symbol": symbol} if symbol else {}
        resp = await self._http.get("/fapi/v1/ticker/price", params=params)
        resp.raise_for_status()
        return resp.json()

    # ── signed endpoints ────────────────────────────────────────────────────

    async def balance(self) -> list[dict]:
        return await self._signed_request("GET", "/fapi/v2/balance", {})

    async def position_risk(self, symbol: str | None = None) -> list[dict]:
        params: dict = {}
        if symbol is not None:
            params["symbol"] = symbol
        return await self._signed_request("GET", "/fapi/v2/positionRisk", params)

    async def open_orders(self, symbol: str | None = None) -> list[dict]:
        params: dict = {}
        if symbol is not None:
            params["symbol"] = symbol
        return await self._signed_request("GET", "/fapi/v1/openOrders", params)

    async def get_order(self, symbol: str, order_id: str) -> dict:
        return await self._signed_request(
            "GET", "/fapi/v1/order", {"symbol": symbol, "orderId": order_id}
        )

    async def place_order(
        self,
        symbol: str,
        side: str,
        order_type: str,
        quantity: str,
        price: str | None = None,
        client_order_id: str | None = None,
    ) -> dict:
        params: dict = {
            "symbol": symbol,
            "side": side,
            "type": order_type,
            "quantity": quantity,
        }
        if price is not None:
            params["price"] = price
        if client_order_id is not None:
            params["newClientOrderId"] = client_order_id
        return await self._signed_request("POST", "/fapi/v1/order", params)

    async def cancel_order(self, symbol: str, order_id: str) -> dict:
        return await self._signed_request(
            "DELETE", "/fapi/v1/order", {"symbol": symbol, "orderId": order_id}
        )
```

```python
"""Binance Futures-spot bookTicker WS subscriber.

We connect to fstream.binance.com (USDT-margined futures), subscribe to
`<symbol>@bookTicker` for each configured symbol, and keep the latest
(bid, ask, ts_ms) per rocky symbol in memory. Auto-reconnect on disconnect.

Calls to `mid(symbol)` raise StaleFeedError if the last update is older
than `stale_after_ms` (default 10000).
"""

from __future__ import annotations

import asyncio
import json
import logging
import time
from dataclasses import dataclass
from decimal import Decimal

import websockets

from .symbol_map import binance_to_rocky, rocky_to_binance

log = logging.getLogger(__name__)

BINANCE_WS_BASE = "wss://fstream.binance.com/stream"


class StaleFeedError(Exception):
    pass


@dataclass(frozen=True)
class Parsed:
    rocky_symbol: str
    bid: Decimal
    ask: Decimal
    ts_ms: int


def parse_book_ticker(payload: dict) -> Parsed | None:
    """Parse a Binance combined-stream payload into Parsed, or None if unknown."""
    data = payload.get("data") or {}
    bsym = data.get("s") or ""
    rocky = binance_to_rocky(bsym)
    if rocky is None:
        return None
    try:
        return Parsed(
            rocky_symbol=rocky,
            bid=Decimal(str(data["b"])),
            ask=Decimal(str(data["a"])),
            ts_ms=int(data.get("E") or data.get("T") or 0),
        )
    except (KeyError, ValueError):
        return None


class BinanceFeed:
    def __init__(self, symbols: list[str], stale_after_ms: int = 10_000):
        self._symbols = symbols
        self._stale = stale_after_ms
        self._cache: dict[str, tuple[Decimal, Decimal, int]] = {}
        self._closed = False

    @staticmethod
    def _dec(x: str) -> Decimal:
        return Decimal(x)

    def mid(self, symbol: str) -> Decimal:
        entry = self._cache.get(symbol)
        if entry is None:
            raise StaleFeedError(f"no data yet for {symbol}")
        bid, ask, ts = entry
        if int(time.time() * 1000) - ts > self._stale:
            raise StaleFeedError(f"feed stale for {symbol}")
        return (bid + ask) / 2

    async def run(self):
        backoff = 1
        while not self._closed:
            try:
                await self._connect_and_consume()
                backoff = 1  # reset after a clean disconnect
            except Exception as e:  # noqa: BLE001 — log + reconnect on anything
                log.warning("binance_feed: %s — reconnecting in %ss", e, backoff)
                await asyncio.sleep(backoff)
                backoff = min(backoff * 2, 30)

    async def _connect_and_consume(self):
        streams = "/".join(
            f"{rocky_to_binance(s).lower()}@bookTicker"
            for s in self._symbols
            if rocky_to_binance(s)
        )
        url = f"{BINANCE_WS_BASE}?streams={streams}"
        log.info("binance_feed: connecting to %s", url)
        async with websockets.connect(url, ping_interval=20, ping_timeout=20) as ws:
            async for raw in ws:
                payload = json.loads(raw)
                p = parse_book_ticker(payload)
                if p:
                    self._cache[p.rocky_symbol] = (p.bid, p.ask, p.ts_ms)

    def close(self):
        self._closed = True
```

```python
"""Hard caps + circuit breaker.

All limits live here. Strategy loops call `check()` at top of each iteration;
if open, they sleep + skip.
"""

from __future__ import annotations

import time
from dataclasses import dataclass


@dataclass
class RiskCaps:
    max_loss_usdc: float = 50.0
    max_notional_usdc: float = 200.0
    max_leverage: int = 10
    api_errors_to_trip: int = 5
    pause_seconds: int = 60
    feed_stale_seconds: int = 10


class CircuitBreaker:
    def __init__(self, caps: RiskCaps):
        self.caps = caps
        self._consecutive_errors = 0
        self._cumulative_pnl = 0.0
        self._opened_at: float | None = None
        self._open_reason: str | None = None

    def is_open(self) -> bool:
        if self._opened_at is None:
            return False
        if time.time() - self._opened_at > self.caps.pause_seconds:
            # auto-close after pause window
            self._opened_at = None
            self._open_reason = None
            self._consecutive_errors = 0
            return False
        return True

    def record_api_success(self):
        self._consecutive_errors = 0

    def record_api_error(self):
        self._consecutive_errors += 1
        if self._consecutive_errors >= self.caps.api_errors_to_trip:
            self._trip(f"api errors >= {self.caps.api_errors_to_trip}")

    def record_realised_pnl(self, delta: float):
        self._cumulative_pnl += delta
        if -self._cumulative_pnl > self.caps.max_loss_usdc:
            self._trip(f"loss > {self.caps.max_loss_usdc}")

    def _trip(self, reason: str):
        self._opened_at = time.time()
        self._open_reason = reason

    @property
    def reason(self) -> str | None:
        return self._open_reason
```

```python
"""MM loop: posts a tight double-sided LIMIT around Binance mid."""

from __future__ import annotations

import asyncio
import logging
from decimal import Decimal

from ..binance_feed import StaleFeedError
from ..risk import CircuitBreaker
from ..symbol_map import rocky_to_binance

log = logging.getLogger(__name__)

SPREAD = Decimal("0.0005")  # 5 bps each side
DRIFT_BPS = Decimal("0.0002")  # 2 bps before we re-quote

MM_QTY_BY_SYMBOL: dict[str, Decimal] = {
    "BTC-PERP": Decimal("0.001"),
    "ETH-PERP": Decimal("0.02"),
}


class MakerLoop:
    def __init__(self, client, feed, symbol: str, circuit: CircuitBreaker, interval_s: float = 2.0):
        self.client = client
        self.feed = feed
        self.symbol = symbol
        self.circuit = circuit
        self.interval_s = interval_s

    async def run(self):
        while True:
            try:
                await self.iterate_once()
            except Exception as e:  # noqa: BLE001
                log.warning("mm[%s] iteration failed: %s", self.symbol, e)
                self.circuit.record_api_error()
            await asyncio.sleep(self.interval_s)

    async def iterate_once(self):
        if self.circuit.is_open():
            return
        try:
            mid = self.feed.mid(self.symbol)
        except StaleFeedError:
            return

        binance_sym = rocky_to_binance(self.symbol) or self.symbol
        target_bid = (mid * (Decimal("1") - SPREAD)).quantize(Decimal("0.01"))
        target_ask = (mid * (Decimal("1") + SPREAD)).quantize(Decimal("0.01"))
        qty = MM_QTY_BY_SYMBOL[self.symbol]

        existing = await self.client.open_orders(symbol=binance_sym)
        self.circuit.record_api_success()

        live_bid = None
        live_ask = None
        for o in existing:
            if o["side"] == "BUY":
                live_bid = o
            elif o["side"] == "SELL":
                live_ask = o

        async def replace(order, target):
            if order is None:
                return False
            drift = abs(Decimal(order["price"]) - target) / target
            if drift > DRIFT_BPS:
                await self.client.cancel_order(symbol=binance_sym, order_id=str(order["orderId"]))
                return True
            return False

        if await replace(live_bid, target_bid):
            live_bid = None
        if await replace(live_ask, target_ask):
            live_ask = None

        if live_bid is None:
            await self.client.place_order(
                symbol=binance_sym,
                side="BUY",
                order_type="LIMIT",
                quantity=str(qty),
                price=str(target_bid),
            )
        if live_ask is None:
            await self.client.place_order(
                symbol=binance_sym,
                side="SELL",
                order_type="LIMIT",
                quantity=str(qty),
                price=str(target_ask),
            )
```

```python
"""Taker loop: fires aggressive LIMIT orders to cross MM's quotes."""

from __future__ import annotations

import asyncio
import logging
import random
from decimal import Decimal

from ..binance_feed import StaleFeedError
from ..risk import CircuitBreaker
from ..symbol_map import rocky_to_binance

log = logging.getLogger(__name__)

TAKER_AGGRESSION = Decimal("0.005")  # cross 50 bps past mid

TAKE_QTY_BY_SYMBOL: dict[str, Decimal] = {
    "BTC-PERP": Decimal("0.0005"),
    "ETH-PERP": Decimal("0.01"),
}


class TakerLoop:
    def __init__(
        self,
        client,
        feed,
        symbol: str,
        circuit: CircuitBreaker,
        base_interval_s: float = 30.0,
        jitter_s: float = 10.0,
        rng=None,
    ):
        self.client = client
        self.feed = feed
        self.symbol = symbol
        self.circuit = circuit
        self.base_interval_s = base_interval_s
        self.jitter_s = jitter_s
        self.rng = rng or random

    async def run(self):
        while True:
            try:
                await self.iterate_once()
            except Exception as e:  # noqa: BLE001
                log.warning("taker[%s] iteration failed: %s", self.symbol, e)
                self.circuit.record_api_error()
            delay = self.base_interval_s + (self.rng.random() * 2 - 1) * self.jitter_s
            await asyncio.sleep(max(1.0, delay))

    async def iterate_once(self):
        if self.circuit.is_open():
            return
        try:
            mid = self.feed.mid(self.symbol)
        except StaleFeedError:
            return

        binance_sym = rocky_to_binance(self.symbol) or self.symbol
        positions = await self.client.position_risk(symbol=binance_sym)
        self.circuit.record_api_success()
        pos_amt = Decimal(positions[0]["positionAmt"]) if positions else Decimal("0")
        mark = Decimal(positions[0]["markPrice"]) if positions else mid
        notional = abs(pos_amt * mark)

        max_notional = Decimal(str(self.circuit.caps.max_notional_usdc))
        if notional > max_notional:
            side = "SELL" if pos_amt > 0 else "BUY"
        else:
            side = self.rng.choice(["BUY", "SELL"])

        qty = TAKE_QTY_BY_SYMBOL[self.symbol]
        if side == "BUY":
            price = (mid * (Decimal("1") + TAKER_AGGRESSION)).quantize(Decimal("0.01"))
        else:
            price = (mid * (Decimal("1") - TAKER_AGGRESSION)).quantize(Decimal("0.01"))

        await self.client.place_order(
            symbol=binance_sym,
            side=side,
            order_type="LIMIT",
            quantity=str(qty),
            price=str(price),
        )
```

```python
"""Entry point. Boots feed, two clients, four strategy loops, gathers."""

from __future__ import annotations

import asyncio
import logging
import signal

import httpx

from .binance_feed import BinanceFeed
from .config import Settings
from .risk import CircuitBreaker, RiskCaps
from .rocky_client import RockyClient
from .strategies.mm import MakerLoop
from .strategies.taker import TakerLoop

log = logging.getLogger(__name__)


async def _main_async(settings: Settings):
    feed = BinanceFeed(settings.symbols)
    mm_http = httpx.AsyncClient(base_url=settings.rocky_fapi_url, timeout=10.0)
    taker_http = httpx.AsyncClient(base_url=settings.rocky_fapi_url, timeout=10.0)
    mm_client = RockyClient(httpc=mm_http, api_key=settings.mm_api_key, api_secret=settings.mm_api_secret)
    taker_client = RockyClient(httpc=taker_http, api_key=settings.taker_api_key, api_secret=settings.taker_api_secret)

    mm_circuit = CircuitBreaker(RiskCaps())
    taker_circuit = CircuitBreaker(RiskCaps())

    tasks = [asyncio.create_task(feed.run(), name="binance_feed")]
    for sym in settings.symbols:
        tasks.append(asyncio.create_task(
            MakerLoop(mm_client, feed, sym, mm_circuit).run(), name=f"mm[{sym}]"
        ))
        tasks.append(asyncio.create_task(
            TakerLoop(taker_client, feed, sym, taker_circuit).run(), name=f"taker[{sym}]"
        ))

    loop = asyncio.get_event_loop()
    stop_event = asyncio.Event()

    def _shutdown():
        log.info("shutdown signal received")
        stop_event.set()

    for sig in (signal.SIGINT, signal.SIGTERM):
        loop.add_signal_handler(sig, _shutdown)

    log.info("rocky-bot started: symbols=%s base=%s", settings.symbols, settings.rocky_fapi_url)
    await stop_event.wait()

    log.info("cancelling %d tasks", len(tasks))
    for t in tasks:
        t.cancel()
    feed.close()
    await asyncio.gather(*tasks, return_exceptions=True)
    await mm_client.aclose()
    await taker_client.aclose()
    log.info("rocky-bot exited cleanly")


def run():
    """Sync entry — invoked by `python -m rocky_bot.main` and `[project.scripts]`."""
    settings = Settings()
    level = logging.DEBUG if settings.log_verbose else logging.INFO
    logging.basicConfig(
        level=level,
        format="%(asctime)s %(levelname)s %(name)s: %(message)s",
    )
    asyncio.run(_main_async(settings))


if __name__ == "__main__":
    run()
```

```bash
#   - .env locally contains the four MM_/TAKER_ key/secret pairs.
#   - Plan A is fully deployed and accepting /fapi requests.
#
# What this does:
#   1. rsync source to ~/rocky-bot on EC2 (preserves .env).
#   2. uv venv + uv pip install -e . on EC2.
#   3. Install systemd user unit + enable lingering so it runs without login.
#   4. systemctl --user restart rocky-bot.
#   5. Tail recent journal as a smoke.
#
# Re-run to redeploy.

set -euo pipefail

SSH_HOST=${SSH_HOST:-ubuntu@13.231.118.218}
SSH_KEY=${SSH_KEY:-~/.ssh/rocky-canton-sandbox.pem}
REMOTE_DIR=${REMOTE_DIR:-rocky-bot}
SCRIPT_DIR="$(cd "$(dirname "${BASH_SOURCE[0]}")" && pwd)"
SSH_OPTS=(-i "${SSH_KEY/#\~/$HOME}" -o StrictHostKeyChecking=accept-new -o ServerAliveInterval=10)

ssh_run() { ssh "${SSH_OPTS[@]}" "$SSH_HOST" "$@"; }

if [ ! -f "$SCRIPT_DIR/.env" ]; then
  echo "FAIL: $SCRIPT_DIR/.env missing. cp .env.example .env and fill in secrets." >&2
  exit 1
fi

echo "==> ensure remote dir + lingering"
ssh_run "mkdir -p ~/${REMOTE_DIR} && loginctl enable-linger ubuntu 2>/dev/null || true"

echo "==> rsync source (preserves remote .env)"
rsync -az --delete \
  --exclude .venv --exclude __pycache__ --exclude '.pytest_cache' --exclude '.ruff_cache' \
  --exclude '*.egg-info' --exclude '.git' --exclude '.env' \
  -e "ssh ${SSH_OPTS[*]}" \
  "$SCRIPT_DIR/" "$SSH_HOST:~/${REMOTE_DIR}/"

echo "==> sync .env to EC2 (will overwrite remote .env)"
scp "${SSH_OPTS[@]}" "$SCRIPT_DIR/.env" "$SSH_HOST:~/${REMOTE_DIR}/.env"

echo "==> install uv if missing"
ssh_run "command -v uv >/dev/null 2>&1 || curl -LsSf https://astral.sh/uv/install.sh | sh"

echo "==> create venv + install"
ssh_run "cd ~/${REMOTE_DIR} && ~/.local/bin/uv venv --python 3.12 && ~/.local/bin/uv pip install -e ."

echo "==> install systemd unit"
ssh_run "mkdir -p ~/.config/systemd/user && cp ~/${REMOTE_DIR}/systemd/rocky-bot.service ~/.config/systemd/user/ && systemctl --user daemon-reload && systemctl --user enable rocky-bot 2>/dev/null || true"

echo "==> restart"
ssh_run "systemctl --user restart rocky-bot"

echo "==> wait 5s and tail journal"
sleep 5
ssh_run "journalctl --user -u rocky-bot --since '10 seconds ago' -n 40 --no-pager"
```
